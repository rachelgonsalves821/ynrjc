# LangUp — Project Requirements
# ================================

# Runtime
# -------
# Node.js >= 18.0.0  (uses native fetch, ES2020+)
# npm    >= 9.0.0

# External Services
# -----------------
# Supabase project  — https://supabase.com
#   Tables: run backend/migrations/001_init_langup_tables.sql
#           then   backend/migrations/002_add_familiarity_fields.sql
#   Auth:   disable "Confirm email" for local dev (Authentication > Providers > Email)
#
# OpenAI account    — https://platform.openai.com
#   Models used: gpt-4o-mini (chat + blend)

# Environment Variables  (backend/.env)
# --------------------------------------
PORT=3000
SUPABASE_URL=https://your-project.supabase.co
SUPABASE_ANON_KEY=your_supabase_anon_key
SUPABASE_SERVICE_KEY=your_supabase_service_role_key
OPENAI_API_KEY=your_openai_api_key

# -----
# 1. cd backend && npm install
